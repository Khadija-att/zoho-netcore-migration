void automation.SandboxOnSellerDeletion1(Int sellerID)
{
info "Processing Lead ID: " + sellerID;
sellerData = zoho.crm.getRecordById("Sellers",sellerID);
info sellerData;
try 
{
	if(sellerData == null)
	{
		info "Failed to fetch Lead data - leadData is null";
		return;
	}
	info "Lead data fetched successfully";
	sellerIdStr = sellerID.toString();
	webhookPayload = Map();
	webhookPayload.put("Seller_Id",sellerIdStr);
	webhookPayload.put("Is_Active","false");
	webhookPayload.put("User_Type","Seller");
	info webhookPayload;
	apiendpoint = "https://qqvnbapdea.execute-api.ap-south-1.amazonaws.com/prod/sandbox-netcore-sync";
	response = invokeurl
	[
		url :apiendpoint
		type :POST
		parameters:webhookPayload.toString()
		headers:{"Content-Type":"application/json"}
	];
	info "API Response: " + response;
	if(response != null)
	{
		info "Webhook sent successfully for Lead ID: " + sellerID;
	}
	else
	{
		info "Failed to send webhook for Lead ID: " + sellerID;
	}
}
catch (error)
{
	info "error: " + error;
}
}
